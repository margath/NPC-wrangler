⚙️ Google Sheets Cloud Setup

Connect this app to a Google Sheet to sync your Campaign Data automatically across devices.

  1. Create a new Google Sheet.
  2. In the top menu, click Extensions > Apps Script.
  3. Delete any default code and paste the script below.
  4. Click Deploy > New deployment. Select type Web app.
  5. Set 'Execute as' to Me and 'Who has access' to Anyone.
  6. Click Deploy, Authorize permissions, and copy the Web app URL.
const NPC_SHEET = "NPCs";
const PC_SHEET = "PCs";
const BACKEND_VERSION = "3.0";

function doGet(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let npcs = [], pcs = [];

    const getSheetData = (sheetName) => {
      let s = ss.getSheetByName(sheetName);
      let arr = [];
      if (s) {
        let data = s.getDataRange().getValues();
        if (data.length > 1) {
          let headers = data[0];
          for (let i = 1; i < data.length; i++) {
            let obj = {};
            headers.forEach((h, idx) => { 
                try { obj[h] = data[i][idx] ? JSON.parse(data[i][idx]) : ""; }
                catch(e) { obj[h] = data[i][idx]; }
            });
            arr.push(obj);
          }
        }
      }
      return arr;
    };

    npcs = getSheetData(NPC_SHEET);
    pcs = getSheetData(PC_SHEET);

    return ContentService.createTextOutput(JSON.stringify({backendVersion: BACKEND_VERSION, npcs: npcs, pcs: pcs})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({error: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    const p = JSON.parse(e.postData.contents);
    if (p.action === 'saveData') {
        const saveSheet = (sheetName, dataArray) => {
            let ss = SpreadsheetApp.getActiveSpreadsheet();
            let s = ss.getSheetByName(sheetName);
            if(!s) s = ss.insertSheet(sheetName);
            s.clear();
            if(dataArray && dataArray.length > 0) {
                let headers = Object.keys(dataArray[0]);
                let rows = [headers];
                dataArray.forEach(item => {
                    rows.push(headers.map(h => JSON.stringify(item[h])));
                });
                s.getRange(1, 1, rows.length, headers.length).setValues(rows);
            }
        };
        
        saveSheet(NPC_SHEET, p.npcs);
        saveSheet(PC_SHEET, p.pcs);
        return ContentService.createTextOutput(JSON.stringify({status: "success"})).setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({error: err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
